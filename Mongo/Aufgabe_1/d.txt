db.Entliehen.aggregate([
  { $group: { _id: "$lnr", anzahl: { $sum: 1 } } },
  { $match: { anzahl: { $gt: 1 } } },
  { $sort: { anzahl: -1 } },
  {
    $lookup: {
      from: "Leser",
      localField: "_id",
      foreignField: "lnr",
      as: "leser"
    }
  },
  { $unwind: "$leser" },
  {
    $project: {
      _id: 0,
      name: "$leser.name",
      anzahl: 1
    }
  }
])
